```python
#!/usr/bin/env python3
"""
IPTV Playlist Downloader
Descarga y combina múltiples playlists M3U
"""

import requests
from datetime import datetime

# Fuentes M3U
SOURCES = [
    "http://paineliptv8.xyz:2082/get.php?username=877884787&password=877884787&type=m3u_plus&output=ts",
    "http://portalshqip.eu:8080/get.php?username=V9qbPfuWL8&password=mYGQj3zBC6&type=m3u_plus",
    "http://dcsport.xyz:8080/get.php?username=NfruMD7aPb&password=VFwC2wcnjn&type=m3u",
    "http://rm.flysat.live:2086/get.php?username=152002563138136&password=QPMzyCcgZj&type=m3u&output=mpegts",
    "http://cineapp.org:8000/get.php?username=mag00:1A:79:6E:C6:A8&password=4kJIEHd1lM&type=m3u_plus",
]

def download_playlist(url, timeout=10):
    """Descarga una playlist M3U"""
    try:
        response = requests.get(url, timeout=timeout)
        response.raise_for_status()
        return response.text
    except Exception as e:
        print(f"Error descargando {url}: {e}")
        return None

def parse_channels(m3u_content):
    """Extrae canales de una playlist M3U"""
    channels = []
    lines = m3u_content.split('\n')
    
    for i, line in enumerate(lines):
        if line.startswith('#EXTINF:'):
            name = line.split(',')[-1].strip() if ',' in line else 'Unknown'
            url = lines[i + 1].strip() if i + 1 < len(lines) else None
            
            if url and not url.startswith('#'):
                channels.append({'name': name, 'url': url})
    
    return channels

def main():
    all_channels = []
    
    print("📥 Descargando playlists...")
    
    for url in SOURCES:
        print(f"  Descargando: {url[:50]}...")
        content = download_playlist(url)
        
        if content:
            channels = parse_channels(content)
            all_channels.extend(channels)
            print(f"  ✓ Encontrados {len(channels)} canales")
    
    # Crear playlist combinada
    output = "#EXTM3U\n"
    output += f"# Generado: {datetime.now().isoformat()}\n"
    output += f"# Total canales: {len(all_channels)}\n\n"
    
    for ch in all_channels:
        output += f"#EXTINF:-1,{ch['name']}\n"
        output += f"{ch['url']}\n"
    
    # Guardar
    filename = f"playlist_combinada_{datetime.now().strftime('%Y%m%d')}.m3u"
    with open(filename, 'w', encoding='utf-8') as f:
        f.write(output)
    
    print(f"\n✅ Playlist guardada: {filename}")
    print(f"📊 Total de canales: {len(all_channels)}")

if __name__ == "__main__":
    main()
```

```typescript
// app/api/iptv/route.ts

import { NextResponse } from 'next/server';

const M3U_SOURCES = [
  {
    name: "Painel IPTV Premium",
    url: "http://paineliptv8.xyz:2082/get.php?username=877884787&password=877884787&type=m3u_plus",
    category: "premium"
  },
  {
    name: "DC Sport",

    category: "sports"
  },
  {

    category: "movies"
  }
];

function parseM3U(content: string) {
  const lines = content.split('\n');
  const channels: any[] = [];
  
  for (let i = 0; i < lines.length; i++) {
    const line = lines[i].trim();
    
    if (line.startsWith('#EXTINF:')) {
      const nameMatch = line.match(/,(.+)$/);
      const logoMatch = line.match(/tvg-logo="([^"]+)"/);
      const groupMatch = line.match(/group-title="([^"]+)"/);
      const idMatch = line.match(/tvg-id="([^"]+)"/);
      
      const url = lines[i + 1]?.trim();
      
      if (url && !url.startsWith('#')) {
        channels.push({
          id: idMatch?.[1] || `ch_${channels.length}`,
          name: nameMatch?.[1] || 'Unknown',
          url: url,
          logo: logoMatch?.[1] || '',
          group: groupMatch?.[1] || 'General'
        });
      }
    }
  }
  
  return channels;
}

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const source = searchParams.get('source') || '0';
  
  try {
    const sourceIndex = parseInt(source);
    const selectedSource = M3U_SOURCES[sourceIndex];
    
    if (!selectedSource) {
      return NextResponse.json({ error: 'Fuente no válida' }, { status: 400 });
    }
    
    const response = await fetch(selectedSource.url);
    const content = await response.text();
    const channels = parseM3U(content);
    
    return NextResponse.json({
      source: selectedSource.name,
      category: selectedSource.category,
      total: channels.length,
      channels: channels
    });
    
  } catch (error) {
    return NextResponse.json({ 
      error: 'Error al obtener canales',
      message: error instanceof Error ? error.message : 'Unknown error'
    }, { status: 500 });
  }
}

export async function POST(request: Request) {
  try {
    const body = await request.json();
    const { url } = body;
    
    if (!url) {
      return NextResponse.json({ error: 'URL requerida' }, { status: 400 });
    }
    
    const response = await fetch(url);
    const content = await response.text();
    const channels = parseM3U(content);
    
    return NextResponse.json({
      total: channels.length,
      channels: channels
    });
    
  } catch (error) {
    return NextResponse.json({ 
      error: 'Error al procesar la playlist'
    }, { status: 500 });
  }
}
```
